function initGa4TokenClient(){
  const cid=localStorage.getItem(GSC_CID_SK)||'';
  if(!cid||!window.google?.accounts?.oauth2)return;
  ga4TokenClient=google.accounts.oauth2.initTokenClient({
    client_id:cid,
    scope:'https://www.googleapis.com/auth/analytics.readonly',
    callback:async(resp)=>{
      if(resp.error){toast('GA4 auth failed: '+resp.error,'err');return;}
      ga4AccessToken=resp.access_token;
      ga4TokenExpiry=Date.now()+(resp.expires_in-60)*1000;
      try{
        const r=await fetch('https://www.googleapis.com/oauth2/v3/userinfo',{headers:{Authorization:'Bearer '+ga4AccessToken}});
        const d=await r.json();ga4UserEmail=d.email||'';
      }catch(e){}
      refreshGa4State();
      toast('✓ Connected to Google Analytics','ok');
    }
  });
}
```

async function ga4FetchReport(propertyId,sd,ed){
  const PAGE=10000;let allRows=[];let offset=0;
  while(true){
    const res=await fetch(`https://analyticsdata.googleapis.com/v1beta/properties/${propertyId}:runReport`,{
      method:'POST',
      headers:{'Authorization':'Bearer '+ga4AccessToken,'Content-Type':'application/json'},
      body:JSON.stringify({
        dateRanges:[{startDate:sd,endDate:ed}],
        dimensions:[{name:'pagePath'}],
        metrics:[{name:'sessions'},{name:'screenPageViews'},{name:'averageSessionDuration'},{name:'bounceRate'}],
        limit:PAGE,offset
      })
    });
    if(!res.ok){const e=await res.json();throw new Error(e.error?.message||'GA4 API error '+res.status);}
    const data=await res.json();
    const rows=data.rows||[];
    allRows=allRows.concat(rows);
    if(rows.length<PAGE)break;
    offset+=PAGE;
  }
  return allRows;
}
```

function ga4MapRows(rawRows,sd,ed){
  const rows=rawRows.map(r=>{
    const dims=r.dimensionValues||[];
    const mets=r.metricValues||[];
    const path=dims[0]?.value||'';
    const sessions=parseFloat(mets[0]?.value||0);
    const views=parseFloat(mets[1]?.value||0);
    const avgDuration=parseFloat(mets[2]?.value||0);
    const bounceRate=parseFloat(mets[3]?.value||0);
    // Build full URL for matching against GSC snapshot rows
    const url=(A.site?('https://'+A.site.replace(/^sc-domain:/,'')):'')
      +(path.startsWith('/')?path:'/'+path);
    return{url,visitors:Math.round(sessions),views:Math.round(views),timeOnPage:avgDuration,bounceRate,active:0,newUsers:0,keyEvents:0};
  }).filter(r=>r.url&&r.visitors>0);
  return{rows,dateRange:sd+' → '+ed,hasViews:true,hasBounce:true,filename:'GA4 Live API'};
}
```

async function pullGa4Live(){
  if(!ga4AccessToken||Date.now()>=ga4TokenExpiry){connectGa4();return;}
  const propId=localStorage.getItem(GA4_PROP_SK)||'';
  if(!propId){toast('Save a GA4 Property ID in Settings first','err');return;}
  const sb=snapB();
  if(!sb){toast('Set a Current (Period B) GSC snapshot first','err');return;}
  const sa=snapA();

  const btn=document.getElementById('ga4-live-pull-btn');
  const prog=document.getElementById('ga4-live-progress');
  const bar=document.getElementById('ga4-live-bar');
  const status=document.getElementById('ga4-live-status');
  if(btn)btn.disabled=true;
  if(prog)prog.style.display='';

  try{
    // Pull Period B (required)
    if(status)status.textContent=`Fetching GA4 data for Current period (${sb.start} → ${sb.end})…`;
    if(bar)bar.style.width='20%';
    const rawB=await ga4FetchReport(propId,sb.start,sb.end);
    if(bar)bar.style.width='50%';
    const parsedB=ga4MapRows(rawB,sb.start,sb.end);
    GA4.B=parsedB;GA4.PS=null;
    try{await sv(GA4_SK+'B',JSON.stringify(parsedB));}catch(e){}
    await pushGa4ToSupabase('B',parsedB);

    // Pull Period A (optional — only if snapshot is set)
    if(sa){
      if(status)status.textContent=`Fetching GA4 data for Baseline period (${sa.start} → ${sa.end})…`;
      if(bar)bar.style.width='60%';
      const rawA=await ga4FetchReport(propId,sa.start,sa.end);
      if(bar)bar.style.width='85%';
      const parsedA=ga4MapRows(rawA,sa.start,sa.end);
      GA4.A=parsedA;GA4.PSA=null;
      try{await sv(GA4_SK+'A',JSON.stringify(parsedA));}catch(e){}
      await pushGa4ToSupabase('A',parsedA);
    }

    if(bar)bar.style.width='100%';
    if(status)status.textContent=`✓ Loaded ${parsedB.rows.length.toLocaleString()} pages`+(sa?` (both periods)`:'')+' — rendering…';

    // Update upload zone status labels
    const bEl=document.getElementById('ga4b-status');
    if(bEl){bEl.innerHTML='✓ <strong>'+parsedB.rows.length.toLocaleString()+' pages</strong> · '+parsedB.dateRange+' <span style="background:rgba(59,158,255,.15);color:var(--blue);border:1px solid rgba(91,159,255,.3);padding:1px 7px;border-radius:4px;font-size:10px;font-weight:600;margin-left:6px">Live API</span>';bEl.className='pp-status loaded';}
    if(sa){const aEl=document.getElementById('ga4a-status');if(aEl){aEl.innerHTML='✓ <strong>'+(GA4.A.rows.length.toLocaleString())+' pages</strong> · '+GA4.A.dateRange+' <span style="background:rgba(59,158,255,.15);color:var(--blue);border:1px solid rgba(91,159,255,.3);padding:1px 7px;border-radius:4px;font-size:10px;font-weight:600;margin-left:6px">Live API</span>';aEl.className='pp-status loaded';}}

    // Hide how-to guide, render table
    const h=document.getElementById('ga4-howto');if(h)h.style.display='none';
    renderPerfView();
    refreshGa4State();
    toast('✓ GA4 data loaded — '+parsedB.rows.length.toLocaleString()+' pages','ok');
    setTimeout(()=>{if(prog)prog.style.display='none';if(bar)bar.style.width='0%';},2000);
  }catch(e){
    toast('GA4 pull failed: '+e.message,'err');
    if(prog)prog.style.display='none';
    if(bar)bar.style.width='0%';
  }
  if(btn)btn.disabled=false;
}
```
